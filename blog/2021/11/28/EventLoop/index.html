EventLoop

November 28, 2021 · 4 min read
setTimeout(function () {
  // 定时器 宏任务
  console.log(1);
}, 0);

new Promise(function executor(resolve) {
  console.log(2);
  for (var i = 0; i < 10000; i++) {
    resolve(); // 微任务
  }
  console.log(3);
}).then(function () {
  console.log(4);
});

console.log(5);

// 答案: 2 3 5 4 1
const promise = new Promise((resolve, reject) => {
  console.log(1);
  resolve(5); // 微任务
  console.log(2);
}).then((res) => {
  console.log(res); // 1.1
});

promise.then(() => {
  console.log(3); // 1.2 微任务
});

console.log(4);

setTimeout(() => {
  console.log(6); // 宏任务
});

// 答案: 1 2 4 5 3 6

常见题 一

async function async1() {
  console.log("async1 start");
  await async2();
  console.log("async1 end"); // 微任务  1.1
}

async function async2() {
  console.log("async2 start");
}

console.log("script start");

setTimeout(function () {
  console.log("setTimeout"); // 宏任务
}, 0);

async1();

new Promise(function (resolve) {
  console.log("promise1");
  resolve();
})
  .then(function () {
    console.log("promise2"); // 微任务
  })
  .then(function () {
    console.log("promise3"); // 微任务
  });

console.log("script end");

// 答案: script start   async1 start  async2 start  promise1
// script end  async1 end  promise2  promise3  setTimeout

变式一

async function async1() {
  console.log("async1 start");
  await async2();
  console.log("async1 end"); // 微任务 1.2
}
async function async2() {
  //async2做出如下更改：
  new Promise(function (resolve) {
    console.log("promise1");
    resolve();
  }).then(function () {
    console.log("promise2"); // 微任务 1.1
  });
}
console.log("script start");

setTimeout(function () {
  console.log("setTimeout"); // 宏任务 2.1
}, 0);
async1();

new Promise(function (resolve) {
  console.log("promise3");
  resolve();
}).then(function () {
  console.log("promise4"); // 微任务 1.3
});

console.log("script end");

// 答案: script start  async1 start  promise1 promise3  script end
// promise2  async1 end  promise4  setTimeout

变式二

async function async1() {
  console.log("async1 start");
  await async2();
  //更改如下：
  setTimeout(function () {
    console.log("setTimeout1"); // 宏任务 2.3
  }, 0);

async function async2() {
  //更改如下：
  setTimeout(function () {
    console.log("setTimeout2"); // 宏任务 2.2
  }, 0);
}
console.log("script start"); //

setTimeout(function () {
  console.log("setTimeout3"); // 宏任务 2.1
}, 0);
async1();

new Promise(function (resolve) {
  console.log("promise1");
  resolve();
}).then(function () {
  console.log("promise2"); //微任务
});

// 答案: script start  async1 start promise1  script end promise2
// setTimeout3  setTimeout2  setTimeout1

变式三

async function a1() {
  console.log("a1 start");
  await a2();
  console.log("a1 end"); // 微任务 1.2
}
async function a2() {
  console.log("a2");
}

console.log("script start");

setTimeout(() => {
  console.log("setTimeout"); // 宏任务 2.1
}, 0);

Promise.resolve().then(() => {
  console.log("promise1"); // 微任务 1.1
});

a1();

let promise2 = new Promise((resolve) => {
  resolve("promise2.then"); // 微任务  1.3
  console.log("promise2");
});

promise2.then((res) => {
  console.log(res);
  Promise.resolve().then(() => {
    console.log("promise3");
  });
});
console.log("script end");

// 答案: script start  a1 start a2 promise2  script end
// promise1  a1 end promise2.then promise3  setTimeout

setImmediate(() => {
  console.log(1);
  setTimeout(() => {
    console.log(2);
  }, 100);
  setImmediate(() => {
    console.log(3);
  });
  process.nextTick(() => {
    console.log(4);
  });
});
process.nextTick(() => {
  console.log(5);
  setTimeout(() => {
    console.log(6);
  }, 100);
  setImmediate(() => {
    console.log(7);
  });
  process.nextTick(() => {
    console.log(8);
  });
});
